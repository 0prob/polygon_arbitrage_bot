### Task 1: Boot Sequence

Create `src/orchestrator/boot.ts` (~100 lines):

```ts
import { createDatabase } from "../infra/db/connection.ts";
import { ensureSchema } from "../infra/db/schema.ts";
import { createRootLogger } from "../infra/observability/logger.ts";
import { createReadClients, createExecutionClient } from "../infra/rpc/client_factory.ts";
import { createHypersyncClient } from "../infra/hypersync/client.ts";
import { DiscoveryService } from "../services/discovery/service.ts";
import { WatcherService } from "../services/watcher/service.ts";
import { HydrationService } from "../services/hydration/service.ts";
import { StrategyEngine } from "../services/strategy/pipeline.ts";
import { ExecutionService } from "../services/execution/service.ts";
import { MempoolService } from "../services/mempool/service.ts";
import type { AppConfig } from "../config/schema.ts";
import type { Logger } from "../infra/observability/logger.ts";

export interface RuntimeContext {
  logger: Logger;
  db: ReturnType<typeof createDatabase>;
  config: AppConfig;
  services: {
    discovery: DiscoveryService;
    watcher: WatcherService;
    hydration: HydrationService;
    strategy: StrategyEngine;
    execution: ExecutionService;
    mempool: MempoolService;
  };
  stateCache: Map<string, unknown>;
  isRunning: boolean;
}

export async function bootApplication(config: AppConfig): Promise<RuntimeContext> {
  const logger = createRootLogger({ level: config.observability.logLevel, pretty: true });
  const db = createDatabase(config.paths.dataDir + "/" + config.paths.dbFile);
  ensureSchema(db);

  const rpcClients = createReadClients(config.rpc.polygonRpcUrls);
  const execClient = createExecutionClient(config.rpc.executionRpcUrl);
  const hypersync = createHypersyncClient({
    url: config.hypersync.url,
    apiToken: config.envioApiToken,
    httpReqTimeoutMillis: config.hypersync.httpReqTimeoutMs,
    maxNumRetries: config.hypersync.maxRetries,
  });

  const stateCache = new Map<string, unknown>();
  const discovery = new DiscoveryService({ logger: logger.child({ service: "discovery" }) });
  const watcher = new WatcherService({ logger: logger.child({ service: "watcher" }) });
  const hydration = new HydrationService({ logger: logger.child({ service: "hydration" }) });
  const strategy = new StrategyEngine({ logger: logger.child({ service: "strategy" }) });
  const execution = new ExecutionService({ logger: logger.child({ service: "execution" }) });
  const mempool = new MempoolService({ logger: logger.child({ service: "mempool" }) });

  logger.info({}, "Application booted");
  return { logger, db, config, services: { discovery, watcher, hydration, strategy, execution, mempool }, stateCache, isRunning: true };
}
```

### Task 4: CLI Entry Point

Create `src/cli/main.ts`:

```ts
import { loadConfig } from "../config/loader.ts";
import { bootApplication } from "../orchestrator/boot.ts";
import { runPassLoop } from "../orchestrator/pass_loop.ts";
import { shutdownApplication } from "../orchestrator/shutdown.ts";

async function main() {
  const config = loadConfig();
  const ctx = await bootApplication(config);
  process.on("SIGINT", () => { ctx.logger.warn({}, "SIGINT received, shutting down"); shutdownApplication(ctx); });
  process.on("SIGTERM", () => { ctx.logger.warn({}, "SIGTERM received, shutting down"); shutdownApplication(ctx); });
  await runPassLoop(ctx);
}

main().catch((err) => { console.error(err); process.exit(1); });
```
